const int mod = 1e9 + 10;
int ksm(int a,int b) {
    int ans = 1;
    while (b > 0) {
        if (b & 1) {
            ans = ans * a % mod;
        }
        a = a*a%mod;
        b >>= 1;
    }
    return ans;
}

int ksm(int a, int b) {
    int ans;
    if (b == 0)
        ans = 1;
    else {
        ans = ksm(a * a, b / 2);
        if (b % 2)
            ans *= a;
    }
    return ans;
}
